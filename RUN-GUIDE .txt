# 🧨 COMPLETE BATCH FILE (COPY PASTE FULL)

```bat
@echo off
title 🚀 FoodSupply Engine - Auto Runner

echo ========================================
echo 🔥 STARTING FOODSUPPLY ENGINE
echo ========================================

echo.
echo 🧹 STEP 1: Freeing Ports...
powershell -Command "Get-NetTCPConnection -LocalPort 8000,9092,3306 -ErrorAction SilentlyContinue | ForEach-Object { Stop-Process -Id $_.OwningProcess -Force }"

echo.
echo 🐳 STEP 2: Starting Docker Containers...
docker start qa-zookeeper
docker start qa-kafka
docker start qa-mysql

timeout /t 5 >nul

echo.
echo 🌐 STEP 3: Starting FastAPI Server...
start cmd /k "title FastAPI Server && uvicorn app.main:app --reload"

timeout /t 3 >nul

echo.
echo ⚙️ STEP 4: Starting Order Service...
start cmd /k "title Order Service && python services/order_service.py"

echo.
echo 🍳 STEP 5: Starting Kitchen Service...
start cmd /k "title Kitchen & Inventory && python services/kitchen_service.py"

echo.
echo 🚚 STEP 6: Starting Delivery Service...
start cmd /k "title Delivery Service && python services/delivery_service.py"

echo.
echo ❌ STEP 7: Starting DLQ Service...
start cmd /k "title DLQ Service && python services/dlq_service.py"

echo.
echo 🚫 STEP 8: Starting Duplicate Service...
start cmd /k "title Duplicate Monitor && python services/duplicate_service.py"

timeout /t 5 >nul

echo.
echo 🌍 STEP 9: Opening Browser...
start http://127.0.0.1:8000/orders

echo.
echo ========================================
echo ✅ SYSTEM STARTED SUCCESSFULLY
echo ========================================

pause
```
